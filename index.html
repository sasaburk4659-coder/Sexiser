import os import base64 import requests GITHUB_TOKEN = os.getenv("GITHUB_TOKEN") REPO_NAME = os.getenv("GITHUB_REPO_NAME", "ai-agent-platform") REPO_DESC = os.getenv("GITHUB_REPO_DESC", "AI Agent Platform") PRIVATE = os.getenv("GITHUB_PRIVATE", "true").lower() == "true" BRANCH = os.getenv("GITHUB_BRANCH", "main") HTML_CONTENT = r""" AI Agent Platform
AI Agent Platform
Состояние: localStorage · agent_platform_v1

Коннекторы

0 подключено·mistral-large-latest
""" if not GITHUB_TOKEN: raise SystemExit("Нет GITHUB_TOKEN в переменных окружения") API = "https://api.github.com" HEADERS = { "Authorization": f"Bearer {GITHUB_TOKEN}", "Accept": "application/vnd.github+json", "X-GitHub-Api-Version": "2022-11-28", } def req(method, url, **kwargs): r = requests.request(method, url, headers=HEADERS, timeout=30, **kwargs) if r.status_code >= 400: try: detail = r.json() except Exception: detail = r.text raise RuntimeError(f"{method} {url} -> {r.status_code}: {detail}") return r def create_repo(): payload = {"name": REPO_NAME, "description": REPO_DESC, "private": PRIVATE, "auto_init": True} return req("POST", f"{API}/user/repos", json=payload).json() def upsert_file(owner, repo, path, content_bytes, message): url = f"{API}/repos/{owner}/{repo}/contents/{path}" sha = None r = requests.get(url, headers=HEADERS, timeout=30) if r.status_code == 200: sha = r.json().get("sha") payload = { "message": message, "content": base64.b64encode(content_bytes).decode("utf-8"), "branch": BRANCH } if sha: payload["sha"] = sha req("PUT", url, json=payload) def main(): repo = create_repo() owner = repo["owner"]["login"] name = repo["name"] upsert_file(owner, name, "index.html", HTML_CONTENT.encode("utf-8"), "Add index.html") upsert_file(owner, name, "README.md", f"# {REPO_DESC}\n\nAutomatically generated.\n".encode("utf-8"), "Add README") print(repo["html_url"]) if __name__ == "__main__": main()
